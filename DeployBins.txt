#!/bin/sh

# change directory and pull with enter password
cd /rost/repo/DEXArb
git pull
# Waiting password for git

# Static Compile
cd src
clang++ -c main.cpp -o main.o
clang++ -c MeasureRunTime.cpp -o MeasureRunTime.o
clang++ -c arbitrage.cpp -o arbitrage.o
clang++ -c input.cpp -o input.o

clang++ -static -O2 -s -o dexarb main.o input.o MeasureRunTime.o arbitrage.o -std=c++17
mv /home/rost/repo/DEXArb/src/dexarb /home/rost/repo/DEXArb/bin/

# Dynamic compile
clang++ -O2 -s -o dexarb main.o input.o MeasureRunTime.o arbitrage.o -std=c++17
mv /home/rost/repo/DEXArb/src/dexarb /home/rost/repo/DEXArb/bin/dexarb2

# Upload to GitHub
cd /home/rost/repo/DEXArb/
git add .
git commit -m "+ obj and binary"
git push origin master
